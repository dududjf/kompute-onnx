30 self.linear_shader = compile_source(f""" 

31#version 450 

32layout(local_size_x={LOCAL_X_1D}) in; 

33 

34layout(std430, set = 0, binding = 0) readonly buffer InBuf {{ float in_buf[]; }}; 

35layout(std430, set = 0, binding = 1) readonly buffer CoefBuf {{ float coef_buf[]; }}; 

36layout(std430, set = 0, binding = 2) writeonly buffer OutBuf {{ float out_buf[]; }}; 

37layout(std430, set = 0, binding = 3) readonly buffer Params {{ 

38 uint feature_size; 

39 uint num_samples; 

40 float rho0; 

41}}; 

42 

43void main() {{ 

44 uint idx = gl_GlobalInvocationID.x; 

45 if (idx >= num_samples) return; 

46  

47 uint in_idx = idx * feature_size; 

48 float s = 0.0; 

49  

50 for (uint k = 0; k < feature_size; ++k) {{ 

51 s += in_buf[in_idx] * coef_buf[k]; 

52 ++in_idx; 

53 }} 

54  

55 out_buf[idx] = s + rho0; 

56}} 

57""")

59 # Common layout for kernel shaders 

60 # 0: InBuf 

61 # 1: SVBuf 

62 # 2: CoefBuf 

63 # 3: OutBuf 

64 # 4: Params 

65 kernel_layout = f""" 

66layout(local_size_x={LOCAL_X_1D}) in; 

67 

68layout(std430, set = 0, binding = 0) readonly buffer InBuf {{ float in_buf[]; }}; 

69layout(std430, set = 0, binding = 1) readonly buffer SVBuf {{ float sv_buf[]; }}; 

70layout(std430, set = 0, binding = 2) readonly buffer CoefBuf {{ float coef_buf[]; }}; 

71layout(std430, set = 0, binding = 3) writeonly buffer OutBuf {{ float out_buf[]; }}; 

72layout(std430, set = 0, binding = 4) readonly buffer Params {{ 

73 uint feature_size; 

74 uint num_samples; 

75 uint n_supports; 

76 float gamma; 

77 float coef0; 

78 float degree; 

79 float rho0; 

80}}; 

81"""

83 self.poly_shader = compile_source(f""" 

84#version 450 

85{kernel_layout} 

86 

87void main() {{ 

88 uint idx = gl_GlobalInvocationID.x; 

89 if (idx >= num_samples) return; 

90  

91 uint base_offset = idx * feature_size; 

92 float s = 0.0; 

93 uint degree_int = uint(degree); 

94 

95 uint sv_offset = 0; 

96 for (uint j = 0; j < n_supports; ++j) {{ 

97 float dot_val = 0.0; 

98 uint in_idx = base_offset; 

99 uint sv_idx = sv_offset; 

100 for (uint k = 0; k < feature_size; ++k) {{ 

101 dot_val += in_buf[in_idx] * sv_buf[sv_idx]; 

102 in_idx++; 

103 sv_idx++; 

104 }} 

105 float kernel_val = dot_val * gamma + coef0;  

106 float powered = 1.0; 

107 for (uint d = 0; d < degree_int; ++d) {{ 

108 powered *= kernel_val; 

109 }} 

110 s += coef_buf[j] * powered; 

111 sv_offset += feature_size; 

112 }} 

113 s += rho0; 

114  

115 out_buf[idx] = s; 

116}} 

117""") 

118 self.rbf_shader = compile_source(f""" 

119#version 450 

120{kernel_layout} 

121 

122void main() {{ 

123 uint idx = gl_GlobalInvocationID.x; 

124 if (idx >= num_samples) return; 

125  

126 float neg_gamma = -gamma; 

127  

128 uint base_offset = idx * feature_size; 

129 float s = 0.0; 

130  

131 uint sv_offset = 0; 

132 for (uint j = 0; j < n_supports; ++j) {{ 

133 float squared_dist = 0.0; 

134 uint in_idx = base_offset; 

135 uint sv_idx = sv_offset; 

136 for (uint k = 0; k < feature_size; ++k) {{ 

137 float diff = in_buf[in_idx] - sv_buf[sv_idx]; 

138 squared_dist += diff * diff; 

139 in_idx++; 

140 sv_idx++; 

141 }} 

142 float kernel_val = exp(neg_gamma * squared_dist); 

143 s += coef_buf[j] * kernel_val; 

144 sv_offset += feature_size; 

145 }} 

146 s += rho0; 

147  

148 out_buf[idx] = s; 

149}} 

150""") 

151 self.sigmoid_shader = compile_source(f""" 

152#version 450 

153{kernel_layout}

155void main() {{ 

156 uint idx = gl_GlobalInvocationID.x; 

157 if (idx >= num_samples) return; 

158  

159 uint base_offset = idx * feature_size; 

160 float s = 0.0; 

161  

162 uint sv_offset = 0; 

163 for (uint j = 0; j < n_supports; ++j) {{ 

164 float dot_val = 0.0; 

165 uint in_idx = base_offset; 

166 uint sv_idx = sv_offset; 

167 for (uint k = 0; k < feature_size; ++k) {{ 

168 dot_val += in_buf[in_idx] * sv_buf[sv_idx]; 

169 in_idx++; 

170 sv_idx++; 

171 }} 

172 float kernel_val = tanh(dot_val * gamma + coef0); 

173 s += coef_buf[j] * kernel_val; 

174 sv_offset += feature_size; 

175 }} 

176 s += rho0; 

177  

178 out_buf[idx] = s; 

179}} 

180""")

188 def run(self, *inputs): 

189 input_tensors = [] 

190 for inp in inputs: 

191 numpy_in = inp.reshape(-1).astype(np.float32) \ 

192 if isinstance(inp, np.ndarray) else np.array(inp, dtype=np.float32) 

193 tensor = self.manager.tensor(numpy_in) 

194 input_tensors.append((tensor, list(inp.shape) if isinstance(inp, np.ndarray) else [])) 

195 

196 updated_algorithms, updated_tensors = [], [] 

197 output_tensor_and_shape = self.fuse(input_tensors, updated_algorithms, updated_tensors) 

198 tensor_out, output_shape = output_tensor_and_shape[0] 

199 

200 if updated_algorithms: 

201 seq = self.manager.sequence() 

202 tensors_to_sync = [t[0] for t in input_tensors] 

203 

204 # Add updated tensors that are NOT the outputs (if any) 

205 # Regressor usually creates one output tensor which is not synced from device (only to local) 

206 # But params tensors need sync. 

207 output_tensor_ids = {id(t) for t, _ in output_tensor_and_shape} 

208 for t in updated_tensors: 

209 if id(t) not in output_tensor_ids: 

210 tensors_to_sync.append(t) 

211 

212 seq.record(kp.OpTensorSyncDevice(tensors_to_sync)) 

213 for alg in updated_algorithms: 

214 seq.record(kp.OpAlgoDispatch(alg)) 

215 seq.record(kp.OpTensorSyncLocal([tensor_out])) 

216 seq.eval() 

217 

218 output = tensor_out.data().reshape(output_shape) 

219 

220 for tensor, _ in input_tensors: 

221 del tensor 

222 del updated_tensors 

223 return [output]

225 def fuse(self, input_tensors: list[tuple[kp.Tensor, list[int]]], updated_algorithms: list[kp.Algorithm], 

226 updated_tensors: list[kp.Tensor]) -> list[tuple[kp.Tensor, list[int]]]: 

227 

228 assert self.post_transform in (None, "NONE"), \ 

229 f"post_transform={self.post_transform!r} not implemented. Only 'NONE' is supported." 

230 

231 tensor_in, in_shape = input_tensors[0] 

232 n_samples = in_shape[0] 

233 feature_size = int(np.prod(in_shape[1:])) 

234 

235 gamma, coef0, degree = ( 

236 (self.kernel_params[0], self.kernel_params[1], self.kernel_params[2]) 

237 if self.kernel_params else (0.0, 0.0, 0) 

238 ) 

239 

240 rho0 = self.rho[0] if self.rho else 0.0 

241 n_supports = self.n_supports if self.n_supports is not None else 0 

242 

243 workgroup = ((n_samples + LOCAL_X_1D - 1) // LOCAL_X_1D, 1, 1) 

244 

245 if n_supports > 0: 

246 kernel_upper = self.kernel_type.upper() 

247 

248 sv_data = self.support_vectors.astype(np.float32).reshape(-1) 

249 coef_data = self.coefficients.astype(np.float32).reshape(-1) 

250 

251 sv_tensor = self.manager.tensor(sv_data) 

252 coef_tensor = self.manager.tensor(coef_data) 

253 tensor_out = self.manager.tensor(np.zeros(n_samples, dtype=np.float32)) 

254 updated_tensors.extend([sv_tensor, coef_tensor, tensor_out]) 

255 

256 # Params: feature_size, num_samples, n_supports, gamma, coef0, degree, rho0 

257 params = np.zeros(7, dtype=np.float32) 

258 params_uint = params.view(np.uint32) 

259 params_uint[0] = feature_size 

260 params_uint[1] = n_samples 

261 params_uint[2] = n_supports 

262 params[3] = gamma 

263 params[4] = coef0 

264 params[5] = float(degree) 

265 params[6] = rho0 

266 

267 if kernel_upper == "LINEAR": 

268 params[3] = 1.0 

269 params[4] = 0.0 

270 params[5] = 1.0 

271 shader = self.poly_shader 

272 elif kernel_upper == "POLY": 

273 shader = self.poly_shader 

274 elif kernel_upper == "RBF": 

275 shader = self.rbf_shader 

276 elif kernel_upper == "SIGMOID": 

277 shader = self.sigmoid_shader 

278 else: 

279 raise ValueError(f"Unexpected kernel_type={self.kernel_type!r}.") 

280 

281 params_tensor = self.manager.tensor_t(params, kp.TensorTypes.device) 

282 self.manager.sequence().record(kp.OpTensorSyncDevice([params_tensor])).eval() 

283 

284 tensors = [tensor_in, sv_tensor, coef_tensor, tensor_out, params_tensor] 

285 

286 updated_algorithms.append(self.manager.algorithm( 

287 tensors, 

288 shader, 

289 workgroup 

290 )) 

291 else: 

292 coef_data = self.coefficients.astype(np.float32).reshape(-1) 

293 

294 coef_tensor = self.manager.tensor(coef_data) 

295 tensor_out = self.manager.tensor(np.zeros(n_samples, dtype=np.float32)) 

296 updated_tensors.extend([coef_tensor, tensor_out]) 

297 

298 # Params: feature_size, num_samples, rho0 (as float) 

299 params = np.zeros(3, dtype=np.float32) 

300 params_uint = params.view(np.uint32) 

301 params_uint[0] = feature_size 

302 params_uint[1] = n_samples 

303 params[2] = rho0 

304 

305 params_tensor = self.manager.tensor_t(params, kp.TensorTypes.device) 

306 self.manager.sequence().record(kp.OpTensorSyncDevice([params_tensor])).eval() 

307 

308 updated_algorithms.append(self.manager.algorithm( 

309 [tensor_in, coef_tensor, tensor_out, params_tensor], 

310 self.linear_shader, 

311 workgroup 

312 )) 

313 

314 shape_out = [n_samples, 1] 

315 return [(tensor_out, shape_out)]
